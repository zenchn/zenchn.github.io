    # 高德地图获取经纬度

    # Demo

    <map-locate v-on:getlocate="setlnglat" :initAccord="formData.latLng"></map-locate>

    setlnglat(data){
  this.formData.longitude = data.lng
  this.formData.latitude = data.lat
  this.formData.latLng = this.formData.longitude + ',' + this.formData.latitude
  this.$refs.baseForm.validateField('latLng')
}
